% Item Commands:
% - \shortitem{title}{year}{description}
% - \longitem{title}{year}{description}  
% - \indentedSubitem{title}{year}{description}
\newcommand{\shortitem}[3]{
  \def\temp{#3}\ifx\temp\empty
  \textbf{#1}
  \else
  \textbf{#1} -- #3
  \fi
  \hfill #2\\[2pt]}

\newcommand{\longitem}[3]{
  \textbf{#1} \hfill #2\\
  \begin{adjustwidth}{0.4cm}{2.2cm}
    #3
    \\[2pt]\end{adjustwidth}}
    
\newcommand{\indentedSubitem}[3]{
  \def\temp{#3}\ifx\temp\empty
  \textbf{#1}
  \else
  \textbf{#1} #3
  \fi
  \hfill #2\\[2pt]}
